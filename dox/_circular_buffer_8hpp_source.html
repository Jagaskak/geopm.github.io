171  template <class type>
172  void CircularBuffer<type>::set_capacity(const unsigned int size)
173  {
174  if (size < m_count) {
175  int size_diff = m_count - size;
176  std::vector<type> temp;
177  //Copy newest data into temporary vector
178  for (unsigned int i = m_head + size_diff; i < ((m_head + m_count) % m_max_size); i = ((i + 1) % m_max_size)) {
179  temp.push_back(m_buffer[i]);
180  }
181  //now re-size and swap out with tmp vector data
182  m_buffer.resize(size);
183  m_buffer.swap(temp);
184  m_count = size;
185  }
186  else {
187  m_buffer.resize(size);
188  }
189  m_head = 0;
190  m_max_size = size;
191  }

193  template <class type>
194  void CircularBuffer<type>::insert(const type value)
195  {
196  if (m_max_size < 1) {
197  throw Exception("CircularBuffer::insert(): Cannot insert into a buffer of 0 size", GEOPM_ERROR_RUNTIME, __FILE__, __LINE__);
198  }
199  if (m_count < m_max_size) {
200  m_buffer[m_count] = value;
201  m_count++;
202  }
203  else {
204  m_buffer[m_head] = value;
205  m_head = ((m_head + 1) % m_max_size);
206  }
